#include <stdio.h>

int main() {
    int choice;
    float balance = 500.00;
    float deposit, withdraw;

    do {
        printf("\n========================\n");
        printf("     SIMPLE ATM MENU    \n");
        printf("========================\n");
        printf("[1] Balance Inquiry\n");
        printf("[2] Deposit\n");
        printf("[3] Withdraw\n");
        printf("[4] Exit\n");
        printf("------------------------\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch(choice) {
            case 1:
                printf("\nYour current balance is: ₱%.2f\n", balance);
                break;

            case 2:
                printf("\nEnter amount to deposit: ₱");
                scanf("%f", &deposit);
                if (deposit > 0) {
                    balance += deposit;
                    printf("Deposit successful! New balance: ₱%.2f\n", balance);
                } else {
                    printf("Invalid deposit amount.\n");
                }
                break;

           case 3:
                printf("\nEnter amount to withdraw: P");
                scanf("%f", &withdraw);
                if (withdraw <= 0) {
                    printf("Invalid withdrawal amount!\n");
                } else if (balance - withdraw < 100) {
                    printf("Withdrawal denied! You must maintain at least P100.00 in your account.\n");
                } else if (withdraw > balance) {
                    printf("Insufficient funds!\n");
                } else {
                    balance -= withdraw;
                    printf("Withdrawal successful! New balance: P%.2f\n", balance);
                }
                break;

            case 4:
                printf("\nThank you for using our ATM. Goodbye!\n");
                break;

            default:
                printf("\nInvalid choice! Please try again.\n");
        }

    } while (choice != 4);

    return 0;
}
