#include <stdio.h>

int main() {
    FILE *file;
    char name[100], section[50];
    char readName[100];

    printf("Enter your Complete Name: ");
    fgets(name, sizeof(name), stdin);

    printf("Enter your Section: ");
    fgets(section, sizeof(section), stdin);

    file = fopen("student.txt", "w");
    if (file == NULL) {
        printf("Error opening file!\n");
        return 1;
    }

    fprintf(file, "Name: %s", name);
    fprintf(file, "Section: %s", section);
    fclose(file);

    printf("\nData has been saved successfully to 'student.txt'!\n");

    file = fopen("student.txt", "r");
    if (file == NULL) {
        printf("Error reading file!\n");
        return 1;
    }

    printf("\n--- Reading from student.txt ---\n");
    while (fgets(readName, sizeof(readName), file) != NULL) {
        printf("%s", readName);
    }

    fclose(file);
    printf("\n-------------------------------\n");

    return 0;
}
