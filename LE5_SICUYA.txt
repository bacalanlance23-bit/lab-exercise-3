#include <stdio.h>

int main() {
    int number;      // variable for user input
    int sum = 0;     // variable to store total sum

    printf("Enter numbers one by one.\n");
    printf("Enter 0 or a negative number to stop.\n\n");

    while (1) { // infinite loop until break
        printf("Enter a number: ");
        scanf("%d", &number);

        if (number > 0) {
            sum += number; // add positive numbers only
        } else {
            break; // total sum and loop on 0 or negative input
        }
    }

    printf("\nThe total sum of all positive numbers entered is: %d\n", sum);

    return 0;
}
