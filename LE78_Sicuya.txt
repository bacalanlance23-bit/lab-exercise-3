#include <stdio.h>

int main() {
    int n, i;
    int numbers[100];
    int largest;
    char text[100];
    int vowel = 0;

    printf("Enter number of elements in the array: ");
    scanf("%d", &n);

    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &numbers[i]);
    }

    largest = numbers[0];
    for(i = 1; i < n; i++) {
        if(numbers[i] > largest) {
            largest = numbers[i];
        }
    }

    printf("\nThe largest element in the array is: %d\n", largest);

    printf("\nEnter a string: ");
    getchar();
    fgets(text, sizeof(text), stdin);

    for(i = 0; text[i] != '\0'; i++) {
        if(text[i] == 'a' || text[i] == 'A' ||
           text[i] == 'e' || text[i] == 'E' ||
           text[i] == 'i' || text[i] == 'I' ||
           text[i] == 'o' || text[i] == 'O' ||
           text[i] == 'u' || text[i] == 'U') {
            vowel++;
        }
    }

    printf("The number of vowels in the string is: %d\n", vowel);

    return 0;
}
